# 0.1 ViewBus — Global Reactive View State Registry

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Create a central ViewRegistry singleton and ViewStateProvider protocol so views can preserve and share state across view switches, replacing ViewRouter's prop-threading model.

**Architecture:** Each view owns a `*ViewState` class implementing a shared `ViewStateProvider` protocol. A central `ViewRegistry` singleton holds provider instances and tracks the active view. Views register on mount and read/write through their provider — state survives unmount/remount. This phase builds the infrastructure + one exemplar (Archive). Remaining views migrate in Phase 1.x.

**Tech Stack:** Svelte 5 runes (`$state`, `$state.raw`, `$derived`), TypeScript strict mode, vitest

```typescript
/**
 * ViewBus Protocol Types — §0.1
 *
 * Defines the shared interface every view must implement
 * to participate in the ViewRegistry system.
 *
 * Protocol surface (per DESIGN_DECISIONS.md §A1):
 *   - selection: current entity selection
 *   - filters: view-specific filter state
 *   - getSnapshot() / restoreSnapshot(): full state serialization
 *
 * This file lives in shared/types/ so both shared/ and features/ can import it.
 */

/**
 * Canonical view identifiers.
 * Maps 1:1 to the 7 user-facing views (keyboard 1-7).
 * Does NOT include deprecated modes ('collections', 'structure', 'trash', 'admin-deps').
 */
export type ViewId =
  | 'archive'
  | 'viewer'
  | 'boards'
  | 'search'
  | 'map'
  | 'timeline'
  | 'metadata';

/**
 * Serializable snapshot of a view's full state.
 * Used for workspace snapshots (Phase 2.2) and view state preservation.
 *
 * `data` is intentionally `Record<string, unknown>` — each view defines
 * its own snapshot shape internally. The registry treats snapshots as opaque.
 */
export interface ViewSnapshot {
  readonly viewId: ViewId;
  readonly version: number;
  readonly data: Record<string, unknown>;
}

/**
 * View-specific filter state.
 * Shape varies per view (Archive: text filter + sort; Map: geo bounds; etc.).
 * Typed as Record to allow cross-view filter propagation in Phase 2.1
 * without the registry needing to know each view's filter shape.
 */
export type ViewFilters = Record<string, unknown>;

/**
 * The protocol every view-level state class must implement.
 *
 * Each view owns a class that implements this interface. The class holds
 * all view-specific reactive state ($state fields). The ViewRegistry
 * holds references to these providers.
 *
 * Consumers read from providers via the registry. Views write to their
 * own provider directly. The registry never mutates provider state.
 *
 * Example implementing class: ArchiveViewState (src/features/archive/stores/)
 */
export interface ViewStateProvider {
  /** Which view this provider manages */
  readonly viewId: ViewId;

  /** Current entity selection in this view (read-only from outside) */
  readonly selection: ReadonlySet<string>;

  /** Replace the selection with a new set of IDs */
  setSelection(ids: ReadonlySet<string>): void;

  /** Current filter state for this view (read-only from outside) */
  readonly filters: Readonly<ViewFilters>;

  /** Serialize full view state to a portable snapshot */
  getSnapshot(): ViewSnapshot;

  /** Restore view state from a snapshot. Implementations should validate viewId. */
  restoreSnapshot(snapshot: ViewSnapshot): void;
}
```

```typescript
/**
 * View Registry — Central View State Manager (§0.1)
 *
 * Holds references to per-view ViewStateProvider instances.
 * Does NOT own or mutate view state — each provider manages its own.
 *
 * Responsibilities:
 *   - Register/unregister view providers
 *   - Track the active view
 *   - Provide cross-cutting reads (activeSelection, activeProvider)
 *
 * Usage:
 *   import { viewRegistry } from '@/src/shared/stores/viewRegistry.svelte';
 *
 *   viewRegistry.register(archiveViewState);
 *   viewRegistry.setActiveView('archive');
 *   viewRegistry.activeProvider?.selection;
 *
 * WARNING: Do NOT destructure — breaks Svelte 5 reactivity.
 */

import type { ViewId, ViewStateProvider } from '@/src/shared/types/viewProtocol';

const EMPTY_SET: ReadonlySet<string> = Object.freeze(new Set<string>());

export class ViewRegistry {
  #providers = $state.raw<Map<ViewId, ViewStateProvider>>(new Map());
  #activeViewId = $state<ViewId>('archive');

  // ── Read-only accessors ──

  /** Currently active view ID */
  get activeViewId(): ViewId {
    return this.#activeViewId;
  }

  /** Provider for the currently active view (if registered) */
  get activeProvider(): ViewStateProvider | undefined {
    return this.#providers.get(this.#activeViewId);
  }

  /** Selection from the active view (empty set if no provider) */
  get activeSelection(): ReadonlySet<string> {
    return this.activeProvider?.selection ?? EMPTY_SET;
  }

  /** List of all registered view IDs */
  get registeredViewIds(): ViewId[] {
    return Array.from(this.#providers.keys());
  }

  // ── Registration ──

  /** Register a view state provider. Replaces existing if same viewId. */
  register(provider: ViewStateProvider): void {
    const next = new Map(this.#providers);
    next.set(provider.viewId, provider);
    this.#providers = next;
  }

  /** Unregister a view state provider. No-op if not registered. */
  unregister(viewId: ViewId): void {
    if (!this.#providers.has(viewId)) return;
    const next = new Map(this.#providers);
    next.delete(viewId);
    this.#providers = next;
  }

  /** Get a registered provider by view ID */
  get(viewId: ViewId): ViewStateProvider | undefined {
    return this.#providers.get(viewId);
  }

  // ── Active View ──

  /** Set the active view. Does not validate registration — view may register later. */
  setActiveView(viewId: ViewId): void {
    this.#activeViewId = viewId;
  }
}

/** Singleton view registry */
export const viewRegistry = new ViewRegistry();
```

```typescript
/**
 * Archive View State — ViewStateProvider implementation (§0.1)
 *
 * Extracts Archive's local state into a store that implements ViewStateProvider.
 * Survives view switches — state persists in the ViewRegistry.
 *
 * Usage:
 *   import { ArchiveViewState } from '@/src/features/archive/stores/archiveViewState.svelte';
 *   const archiveState = new ArchiveViewState();
 *   viewRegistry.register(archiveState);
 */

import type {
  ViewId,
  ViewSnapshot,
  ViewFilters,
  ViewStateProvider,
} from '@/src/shared/types/viewProtocol';

// ── Archive-specific types (match ArchiveView.svelte conventions) ──

export type ArchiveViewMode = 'grid' | 'list' | 'grouped';
export type ArchiveSortBy = 'name' | 'date' | 'size';
export type ArchiveSortDirection = 'asc' | 'desc';

const SNAPSHOT_VERSION = 1;

export class ArchiveViewState implements ViewStateProvider {
  readonly viewId: ViewId = 'archive';

  // ── Reactive state ──

  #viewMode = $state<ArchiveViewMode>('grid');
  #filter = $state('');
  #sortBy = $state<ArchiveSortBy>('name');
  #sortDirection = $state<ArchiveSortDirection>('asc');
  #groupByManifest = $state(false);
  #selectedIds = $state<Set<string>>(new Set());
  #activeItemId = $state<string | null>(null);

  // ── ViewStateProvider: selection ──

  get selection(): ReadonlySet<string> {
    return this.#selectedIds;
  }

  setSelection(ids: ReadonlySet<string>): void {
    this.#selectedIds = new Set(ids);
  }

  // ── ViewStateProvider: filters ──

  get filters(): Readonly<ViewFilters> {
    return {
      textFilter: this.#filter,
      sortBy: this.#sortBy,
      sortDirection: this.#sortDirection,
      groupByManifest: this.#groupByManifest,
    };
  }

  // ── ViewStateProvider: snapshot ──

  getSnapshot(): ViewSnapshot {
    return {
      viewId: this.viewId,
      version: SNAPSHOT_VERSION,
      data: {
        viewMode: this.#viewMode,
        filter: this.#filter,
        sortBy: this.#sortBy,
        sortDirection: this.#sortDirection,
        groupByManifest: this.#groupByManifest,
        selectedIds: Array.from(this.#selectedIds),
        activeItemId: this.#activeItemId,
      },
    };
  }

  restoreSnapshot(snapshot: ViewSnapshot): void {
    if (snapshot.viewId !== this.viewId) {
      throw new Error(
        `ArchiveViewState: snapshot viewId mismatch — expected "${this.viewId}", got "${snapshot.viewId}"`
      );
    }
    const d = snapshot.data;
    this.#viewMode = (d.viewMode as ArchiveViewMode) ?? 'grid';
    this.#filter = (d.filter as string) ?? '';
    this.#sortBy = (d.sortBy as ArchiveSortBy) ?? 'name';
    this.#sortDirection = (d.sortDirection as ArchiveSortDirection) ?? 'asc';
    this.#groupByManifest = (d.groupByManifest as boolean) ?? false;
    this.#selectedIds = new Set(Array.isArray(d.selectedIds) ? (d.selectedIds as string[]) : []);
    this.#activeItemId = (d.activeItemId as string | null) ?? null;
  }

  // ── Archive-specific reads ──

  get viewMode(): ArchiveViewMode { return this.#viewMode; }
  get filter(): string { return this.#filter; }
  get sortBy(): ArchiveSortBy { return this.#sortBy; }
  get sortDirection(): ArchiveSortDirection { return this.#sortDirection; }
  get groupByManifest(): boolean { return this.#groupByManifest; }
  get activeItemId(): string | null { return this.#activeItemId; }

  // ── Archive-specific mutations ──

  setViewMode(mode: ArchiveViewMode): void { this.#viewMode = mode; }
  setFilter(filter: string): void { this.#filter = filter; }
  setSortBy(sortBy: ArchiveSortBy): void { this.#sortBy = sortBy; }
  setSortDirection(dir: ArchiveSortDirection): void { this.#sortDirection = dir; }
  toggleSortDirection(): void {
    this.#sortDirection = this.#sortDirection === 'asc' ? 'desc' : 'asc';
  }
  setGroupByManifest(group: boolean): void { this.#groupByManifest = group; }
  setActiveItemId(id: string | null): void { this.#activeItemId = id; }

  // ── Selection helpers (beyond ViewStateProvider minimum) ──

  toggleSelection(id: string): void {
    const next = new Set(this.#selectedIds);
    if (next.has(id)) {
      next.delete(id);
    } else {
      next.add(id);
    }
    this.#selectedIds = next;
  }

  clearSelection(): void {
    this.#selectedIds = new Set();
  }

  selectRange(ids: string[]): void {
    const next = new Set(this.#selectedIds);
    ids.forEach(id => next.add(id));
    this.#selectedIds = next;
  }
}
```
